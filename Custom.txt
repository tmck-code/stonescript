?hp < 30
  activate potion

?loc = caves
  equipL sword
  equipR shield
  ?foe = boss
    equip crossbow

?loc = fungus_forest
  ?pickup.distance < 10
    equipL star
  ?foe boss
    equipL fire wand
    equipR poison wand
  :
    ?foe.distance > 30
      equipL tri
    :
      equipL poison wand
      equipR fire wand

?loc = undead_crypt
  ?pickup.distance < 10
    equipL star
  ?foe = boss
    equipL crossbow
  :
    ?foe.distance > 30
      equipL tri
    :
      equipL vigor wand
      equipR poison wand

?loc = halls
 equipL poison wand
 equipR vigor wand

?loc = rocky
  ?harvest.distance < 3
    >using shovel
    equip shovel
  :
    >using tri
    equipL tri
  ?hp < 31
    activate potion
  ?foe = boss
    equipL fire wand
    equipR poison wand

?loc = mine
  ?pickup.distance < 10
    equipL star
  :?foe ! boss
    ?foe.distance > 30
      equipL tri
    :?foe.distance <= 25 & foe.distance >= 20
      equipR dashing
    :
      equipR ice wand
      equipL aether wand
  :
    equipL aether wand
    equipR ice wand

?loc = ridge
  ?pickup.distance < 10
    equipL star
  :?foe ! boss & foe.distance > 30
    equipL tri
  :
    equipL poison wand
    equipR fire wand
