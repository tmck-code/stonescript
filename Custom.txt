// increase loot
func LootStar()
  ?pickup.distance < 10
    equipL star
    equipR tri
    >o-6,3,#yellow,Loot!!
    return true
  return false

// auto-use potion when hp < 30
func AutoPotion()
  ?hp < 30
    activate potion
