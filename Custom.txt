// import all shared strategies & logging utility
var strats = import SS/Strats
import SS/Log

func ToggleRemoteCache()
  // Use the "up" key to toggle remote caching of scripts on & off
  ?key = upBegin
    sys.cacheRemoteFiles = !sys.cacheRemoteFiles
    ?sys.cacheRemoteFiles
      >Remote caching ENABLED
    :
      >Remote caching DISABLED

// enable auto-heal (via potions)
strats.AutoPotion()
// enable bonus loot with star stone
strats.LootStar()
// enable remote cache toggle
ToggleRemoteCache()
