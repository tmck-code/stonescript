func Fmt(data)
  var strParts = []
  for i = 0 .. data.Count()-1
    strParts.Add(
    ^  string.Format("{0}", data[i])
    ^)
  return string.Join("/", strParts)

func FmtFoes()
  ?foe.Count > 0
    return string.Format("n:{0}, {1}m - {2}/{3}hp", foe.Count, foe.distance, foe.id, foe.hp)

func LogTop(l)
  for i = 0 .. l.Count()-1
    >`0,@i+1@,@l[i][0]@: @Fmt(l[i][1])@

func LogStrat(y, l, colour)
  >`0,@y@,@colour@,strat: @l@
