var log = import SS/Log
var strats = import SS/Strats

func FungusForest()

  ?foe = boss
    // strats.BigBlade(1)
    equip heavy_hammer
    strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
    strats.UseTwoHandedWeaponIfReady("bardiche", 10)
    // equipL fire wand
    // equipR poison wand
  :
    ?!strats.BeFast(true, true)
      // strats.BigBlade(6)
      equipL poison wand
      equipR fire wand

?loc = fungus_forest
  FungusForest()
