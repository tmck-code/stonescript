var log = import SS/Log
var strats = import SS/Strats

func FungusForest()
  log.LogStrat(4, "FungusForest", "#green")
  strats.UseAvailableTwoHandedWeapon()
  strats.BigBlade(18)

  ?foe = boss
    strats.BigBlade(1)
    equipL fire wand
    equipR poison wand
  :
    ?!strats.BeFast(true, true)
      equipL poison wand
      equipR fire wand

?loc = fungus_forest
  FungusForest()
