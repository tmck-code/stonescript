var log = import SS/Log
var strats = import SS/Strats

func FungusForest()
  log.LogStrat(4, "FungusForest", "#green")

  ?foe = boss
    ?!strats.BeFast(true, true)
      ?foe.distance >= 5
        equip poison crossbow
      :
        ?foe.distance <= 3
          ?item.GetCooldown("heavy_hammer") = 0 & item.CanActivate()
            equip heavy hammer +10
            activate R
        :
          equipR compound shield
          equipL poison sword
  :?!strats.BeFast(true, false)
    ?foe.distance >= 17
      equip poison crossbow
    :?foe.distance >= 10
      equipR poison wand
      equipL fire wand
    :
      equipL poison sword
      equipR compound

?loc = fungus_forest
  FungusForest()
