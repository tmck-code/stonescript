var log = import SS/Log
var strats = import SS/Strats

func FungusForest()
  log.LogStrat(4, "FungusForest", "#green")

  ?foe = boss
    strats.BigBlade(1)
    equip heavy_hammer
    strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
    // equipL fire wand
    // equipR poison wand
  :
    ?!strats.BeFast(true, true)
      strats.BigBlade(18)
      equipL poison wand
      equipR fire wand

?loc = fungus_forest
  FungusForest()
