var log = import SS/Log
var strats = import SS/Strats

func HauntedHalls()
  log.LogStrat(4, "HauntedHalls", "#green")

  ?foe = boss
    strats.BigBlade(1)
    strats.UseTwoHandedWeaponIfReady("bardiche", 10)
    equip heavy hammer
    strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
    equipL vigor wand
    equipR poison wand
    >o-6,3,#blue,Lets boss!
  :
    ?buffs.GetTime("pick_pocket") < 100 & foe ! immune_to_physical 
      equip skeleton_arm
    :?!strats.BeFast(true, true)
      // strats.Pickpocket()
      >o-6,3,#red,Lets fight!
      strats.BigBlade(7)
      ?foe.distance > 16
        equipL poison wand
        equipR vigor wand
      :?foe.distance <= 16 & foe.distance >= 11
        equipR dash
        strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
      :?!foe = immune_to_physical
        equip heavy hammer
        strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
      :
        // strats.Pickpocket()
        equipL poison wand
        equipR vigor wand

?loc = halls
  HauntedHalls()
