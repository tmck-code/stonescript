var log = import SS/Log
// var info = import Data/Info
var strats = import SS/Strats

var BOSS = false
var i = 0

func HauntedHalls()
  ?loc.loop
    BOSS = false
  log.LogStrat(4, "HauntedHalls", "#green")

  >`0,18,#green,@log.Get("halls:enemy_type") - @log.Get("halls:vulnerability")@

  ?foe = boss | BOSS = true
    BOSS = true
    >o-6,3,#blue,Lets boss! @BOSS@
    ?foe.distance > 25 & foe ! ghost
      equip vigor crossbow
    :
      equipL vigor wand
      equipR poison wand
  :
    ?!strats.BeFast(true, true)
      >o-6,3,#red,Lets fight! @BOSS@

      ?foe.distance > 20 & foe ! ghost
        equip vigor crossbow

      :?foe ! immune_to_physical
        strats.UseWeaponIfReady("heavy_hammer", 10)
        equipL vigor sword
        equipR war hammer *9
      :
        equipL poison wand
        equipR vigor wand

?loc = halls
  HauntedHalls()
