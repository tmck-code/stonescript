var log = import SS/Log
var strats = import SS/Strats

func HauntedHalls()
  log.LogStrat(4, "HauntedHalls", "#green")

  ?foe = boss
    strats.BigBlade(1)
    strats.UseTwoHandedWeaponIfReady("bardiche", 10)
    equip heavy hammer
    strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
    equipL vigor wand
    equipR poison wand
    >o-6,3,#blue,Lets boss!
  :
    ?!strats.BeFast(true, true)
      >o-6,3,#red,Lets fight!
      ?(foe.hp + foe.armor) <= 15
        strats.Pickpocket()
      strats.BigBlade(7)
      ?foe.distance > 16
        equipL poison wand
        equipR vigor wand
        strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
      :?!foe = immune_to_physical
        equip heavy hammer
        strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
      :
        equipL poison wand
        equipR vigor wand

?loc = halls
  HauntedHalls()
