var log = import SS/Log
var strats = import SS/Strats

func HauntedHalls()
  log.LogStrat(4, "HauntedHalls", "#green")

  ?foe = boss | BOSS = true
    BOSS = true
    >o-6,3,#blue,Lets boss! @BOSS@
    strats.MaskOfTheLost()
    strats.BigBlade(1)
    ?foe.distance > 25 & foe ! ghost
      equip vigor crossbow
    :
      equipL vigor wand
      equipR poison wand
  :
    ?!strats.BeFast(true, true)
      >o-6,3,#red,Lets fight! @BOSS@
      strats.MaskOfTheLost()
      strats.BigBlade(10)

      ?foe.distance > 20 & foe ! ghost
        equip vigor crossbow

      :?foe ! immune_to_physical
        strats.UseTwoHandedWeaponIfReady("heavy_hammer", 10)
        equipL vigor sword
        equipR war hammer *9
      :
        equipL poison wand
        equipR vigor wand

var BOSS = false
