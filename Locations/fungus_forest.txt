var log = import SS/Log
var strats = import SS/Strats

func FungusForest()
  log.LogStrat(4, "FungusForest", "#green")
  strats.LootStar()

  ?foe = boss
    equipL fire wand
    equipR poison wand
  :
    ?!strats.Speedy(30)
      equipL poison wand
      equipR fire wand

?loc = fungus_forest
  FungusForest()
