var log = import SS/Log
var strats = import SS/Strats

func Caves()
  log.LogStrat(4, "Caves", "#green")

  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss
    equipL fire wand
    equipR ice wand
  :
    ?!strats.BeFast(true, true)
      ?foe.distance > 12
        equip ice staff
      :?foe.distance < 5
        equipL ice hammer
        equipR ice wand
      :
        equipL vigor wand
        equipR ice wand

?loc = caves
  Caves()
