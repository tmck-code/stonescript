var log = import SS/Log
var strats = import SS/Strats

func Caves()
  log.LogStrat(4, "Caves", "#green")

  ?foe = boss
    >o-6,3,#blue,Lets boss!
    strats.BigBlade(1)
    equipL vigor wand
    equipR ice wand
  :
    >o-6,3,#red,Lets fight!
    ?!strats.BeFast(true, true)
      ?(foe.hp + foe.armor) <= 15
        strats.Pickpocket()
      equipL vigor wand
      equipR ice wand

?loc = caves
  Caves()
