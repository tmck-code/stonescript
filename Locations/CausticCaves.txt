var log = import SS/Log
var strats = import SS/Strats

func Caves()
  log.LogStrat(4, "Caves", "#green")
  strats.UseAvailableTwoHandedWeapon()
  strats.BigBlade(5)

  ?foe = boss
    // equip skeleton_arm
    strats.BigBlade(1)
    equipL vigor wand
    equipR ice wand
    >o-6,3,#blue,Lets boss!
  :
    equipL vigor wand
    equipR poison wand
    ?!strats.BeFast(true, true)
      equipL vigor wand
      equipR poison wand
      >o-6,3,#red,Lets fight!

?loc = caves
  Caves()
