var log = import SS/Log
var strats = import SS/Strats

func Caves()
  log.LogStrat(4, "Caves", "#green")

  ?strats.canUseBladeDS(2)
    strats.useBladeDS()
    return

  ?foe = boss
    equipL fire wand
    equipR ice wand
    return

  ?strats.BeFast(true, true)
    return

  ?(foe.hp + foe.armor) <= 10
    strats.Pickpocket()

  ?foe.distance > 12
    equipL ice crossbow
    return

  equipL stone wand
  equipR ice wand

?loc = caves
  Caves()
