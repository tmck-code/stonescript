var log = import SS/Log
var strats = import SS/Strats

func Caves()
  log.LogStrat(4, "Caves", "#green")

  ?foe = boss
    equipL stone wand
    equipR ice wand
  :
    ?!strats.BeFast(true, true)
      equipL stone wand
      equipR ice wand

      ?(foe.hp + foe.armor) <= 10
        strats.Pickpocket()

?loc = caves
  Caves()
