var log = import SS/Log
var strats = import SS/Strats

func BoilingMine()
  log.LogStrat(0, "BoilingMine", "#green")
  strats.BigBlade(16)
  strats.BeFast(true, true)

  ?foe = boss
    >o-6,3,#blue,Lets fight!
    strats.MaskOfTheLost()
    strats.BigBlade(1)
    strats.UseAvailableTwoHandedWeapon()

    equipR aether wand
    equipL poison wand
  :
    ?!strats.BeFast(true, false)
      >o-6,3,#red,Lets fight!
      strats.MaskOfTheLost()
      strats.BigBlade(10)
      strats.UseAvailableTwoHandedWeapon()

      ?((foe = slave_master | foe = elemental) & foe.distance > 10) | foe.distance > 20
        equip fire crossbow
      :
        equipL big aether sword
        equipR bashing
?loc = mine
  BoilingMine()
