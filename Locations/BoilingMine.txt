var log = import SS/Log
var strats = import SS/Strats

func BoilingMine()
  log.LogStrat(0, "BoilingMine", "#green")
  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss
    ?log.CheckDebuff("debuff_move_speed")
      equipL aether wand
      equipR aether wand
    :
      equip aether crossbos
  :
    ?!strats.BeFast(true, true)
      strats.UseAvailableTwoHandedWeapon()

      ?((foe = slave_master | foe = elemental) & foe.distance > 10)
        equip aether crossbow
      :?foe.distance > 20
        equip aether crossbow
      :?foe.distance > 10
        equipL aether wand
        equipR vigor wand
      :
        equipL big aether sword
        equipR compound shield

?loc = mine
  BoilingMine()
