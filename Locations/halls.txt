var log = import SS/Log
var strats = import SS/Strats

func Halls()
  log.LogStrat(4, "Halls", "#green")

  strats.LootStar()
  ?pickup.distance < 10
    equipL star
    >o-6,3,#yellow,Loot!!

  ?foe = boss
    equipL vigor wand
    equipR poison wand
    >o-6,3,#blue,Lets boss!
  :
    ?!strats.Speedy(30)
      equipL poison wand
      equipR vigor wand
      >o-6,3,#red,Lets fight!

?loc = halls
  Halls()
