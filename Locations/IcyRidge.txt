var log    = import SS/Log
var strats = import SS/Strats

func IcyRidge()
  log.LogStrat(4, "IcyRidge", "#green")

  ?foe = boss
    strats.BigBlade(1)
    strats.UseAvailableTwoHandedWeapon()
    // equips both fire wands
    loadout 1
    ?!strats.Speedy(25)
      ?foe = ice_elemental | foe = ice_wall
        equip fire crossbow
      :
        strats.BigBlade(4)
        strats.UseAvailableTwoHandedWeapon()
        // equips both fire wands
        loadout 1
        >o-6,3,#red,Lets fight!

?loc = icy_ridge
  IcyRidge()
