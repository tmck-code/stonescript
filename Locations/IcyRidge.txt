var log    = import SS/Log
var strats = import SS/Strats

func IcyRidge()
  log.LogStrat(4, "IcyRidge", "#green")
  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss
    strats.UseAvailableTwoHandedWeapon()
    equipL fire hammer
    equipR compound shield
    // ?foe.distance > 12
    //   strats.UseWeaponIfDist("fissure", 18)
    //   equip fire crossbow
    // :?foe.distance <= 5
    //   strats.UseAvailableTwoHandedWeapon()
    //   equipL fire hammer
    //   equipR compound shield
    // :
    //   equipL fire wand *9
      // equipR fire wand *10
  :
    ?!strats.BeFast(true, true)
      ?foe = ice_elemental | foe.distance > 12
        equip fire crossbow
      :?foe.distance <= 5
        strats.UseAvailableTwoHandedWeapon()
        equipL fire hammer
        equipR compound shield
      :
        equipL fire wand *9
        equipR fire wand *10

?loc = icy_ridge
  IcyRidge()
