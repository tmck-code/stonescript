var log    = import SS/Log
var strats = import SS/Strats

func IcyRidge()
  log.LogStrat(4, "IcyRidge", "#green")
  strats.LootStar()

  ?foe = boss
    strats.BigBlade(1)
    strats.UseAvailableTwoHandedWeapon()
    equip loadout 1 // equips both fire wands
  :?foe = ice_wall
    equip fire crossbow
  :?!strats.Speedy(25)
    ?foe = ice_elemental & foe.distance <= 22
      equip fire crossbow
    :
      strats.BigBlade(4)
      strats.UseAvailableTwoHandedWeapon()
      equip loadout 1 // equips both fire wands
      >o-6,3,#red,Lets fight!

?loc = icy_ridge
  IcyRidge()
