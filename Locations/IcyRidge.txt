var log    = import SS/Log
var strats = import SS/Strats

func IcyRidge()
  log.LogStrat(4, "IcyRidge", "#green")
  strats.LootStar()

  ?foe = boss
    strats.BigBlade(1)
    strats.UseAvailableTwoHandedWeapon()
    // equip fire staff
    equipL fire wand +6
    equipR fire wand +7
  :?foe = ice_wall
    equip fire crossbow
  :?!strats.Speedy(25)
    ?foe = ice_elemental & foe.distance <= 22
      equip fire crossbow +6
    :
      strats.BigBlade(4)
      strats.UseAvailableTwoHandedWeapon()
      equipL fire wand +6
      equipR fire wand +7
      >o-6,3,#red,Lets fight!

?loc = icy_ridge
  IcyRidge()
