var log = import SS/Log
var strats = import SS/Strats
import SS/Checks/CanUse

func BoilingMine()
  log.LogStrat(1, "BoilingMine", "#green")

  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss & foe = bomb_cart
    ?strats.Speedy()
      return
    equip aether crossbow
    return

  ?foe = boss
    ?foe.state = 32
      ?foe.distance > 10
        equipL tri
        return
    ?foe.state = 33
      ?foe.distance > 10
        equipL tri
        return
      ?strats.UseWeaponIfReady("bardiche", 10)
        return
      ?strats.UseWeaponIfReady("heavy_hammer", 10)
        activate R
        return
      equip heavy hammer
      // equipL aether sword
      // equipR stone sword
      return

    ?foe.armor <= 500
      ?strats.MaskOfTheLost()
        return
      ?strats.UseWandAbility("plague", "R")
        return
      ?strats.UseWandAbility("explosive", "L")
        return
      equip repeating crossbow
      return
    equip aether crossbow
    return

  ?strats._Speedy(22, true)
    return

  ?(foe = slave_master | foe = elemental) | foe.distance > 12
    equip aether crossbow
    return

  ?foe.count > 12
    ?armor > 14
      strats.UseWandAbility("calamity", "L")
      return
    equipL aether wand
    equipR vigor wand
    return

  ?foe.distance > 12
    equipL aether wand
    equipR vigor wand
    return

  ?strats.Dash()
    return

  equipL aether sword +8
  equipR aether hammer +8

?loc = mine
  BoilingMine()
