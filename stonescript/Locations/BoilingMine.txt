var log = import SS/Log
var strats = import SS/Strats
var CanUse = import SS/Checks/CanUse
var gameEngine = import SS/GameEngine

func Override()
  equip aether staff
  ?CanUse.AbilityOfStaff("grasping")
    activate R
    return true
  return true

func BoilingMine()
  log.LogStrat(1, "BoilingMine", "#green")

  ?strats.canUseBladeDS(4)
    strats.useBladeDS()

  ?foe = boss & foe = bomb_cart
    ?strats.Speedy()
      return
    equip aether crossbow
    return

  ?foe = boss
    ?foe.state = 32
      ?foe.distance > 10
        equipL tri
        return
    ?foe.state = 33
      ?foe.distance > 10
        equipL tri
        return
      // ?strats.UseWeaponIfReady("bardiche", 10)
      //   return
      // ?strats.UseWeaponIfReady("heavy_hammer", 10)
      //   activate R
      //   return
      // equip heavy hammer
      equipL aether sword +8
      equipR aether hammer +8
      return

    ?foe.armor <= 500
      // ?strats.MaskOfTheLost()
      //   return
      // ?strats.UseWandAbility("plague", "R")
      //   return
      // ?strats.UseWandAbility("explosive", "L")
      //   return
      equip aether crossbow
      return
    equip aether crossbow
    return

  ?strats._Speedy(22, false)
    return

  ?foe.count <= 4
    ?strats._Dash(11, 20)
      return

  ?(foe = slave_master | foe = elemental) | foe.count > 8
    equip aether crossbow
    return

  ?strats.Dash()
    return

  // ?foe.count > 12
  //   ?armor > 14
  //     strats.UseWandAbility("calamity", "L")
  //     return
  //   equipL aether wand
  //   equipR vigor wand
  //   return

  // ?foe.distance > 18
  //   equipL aether wand
  //   equipR vigor wand
  //   return


  equipL aether sword +8
  equipR aether hammer +8

?loc = mine
  ?gameEngine.OverrideEnabled() & Override()
    return
  BoilingMine()
