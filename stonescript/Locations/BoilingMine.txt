var log = import SS/Log
var strats = import SS/Strats
var CanUse = import SS/Checks/CanUse
var gameEngine = import SS/GameEngine

func Override()
  equip aether staff
  ?CanUse.AbilityOfStaff("grasping")
    activate R
    return true
  return true

func BoilingMine()
  log.LogStrat(1, "BoilingMine", "#green")

  ?strats.canUseBladeDS(3)
    strats.useBladeDS()

  ?foe = boss & foe = bomb_cart
    ?strats.Speedy()
      return
    equip aether crossbow
    return

  ?foe = boss
    activate potion
    ?foe.state = 32
      ?foe.distance > 10
        equipL tri
        return
    ?foe.state = 33
      ?foe.distance > 10
        equipL tri
        return
      // ?strats.UseWeaponIfReady("bardiche", 10)
      //   return
      // ?strats.UseWeaponIfReady("heavy_hammer", 10)
      //   activate R
      //   return
      // equip heavy hammer
      equipL big aether sword
      equipR aether hammer
      return

    ?foe.armor <= 500
      // ?strats.MaskOfTheLost()
      //   return
      // ?strats.UseWandAbility("plague", "R")
      //   return
      // ?strats.UseWandAbility("e\xplosive", "L")
      //   return
      equip aether crossbow
      return
    equip aether crossbow
    return

  ?strats._Speedy(25, false)
    return

  ?foe.distance > 15 & foe.count > 6 // | (foe = slave_master | foe = elemental)
    equip aether crossbow
    return

  ?foe.count <= 8 & !((foe = mine_walker | foe = big_slave_master) & foe.hp > 30)
    ?strats._Dash(11, 18)
      return

  ?foe.count < 4
    ?strats._Dash(11, 18)
      return

  ?foe.distance > 15 & foe.count > 6 // | (foe = slave_master | foe = elemental)
    equip aether crossbow
    return
  
  ?foe ! mine_walker & foe ! big_slave_master
    ?strats._Dash(11, 18)
      return

  ?foe.distance > 15 & foe.count > 10
    equip aether crossbow
    return
  
  // ?strats.Dash()
  //   return

  // ?foe.count > 12
  //   ?armor > 14
  //     strats.UseWandAbility("calamity", "L")
  //     return
  //   equipL aether wand
  //   equipR vigor wand
  //   return

  // ?foe.distance > 18
  //   equipL aether wand
  //   equipR vigor wand
  //   return

  equipL big aether sword
  equipR aether hammer

?loc = mine
  ?gameEngine.OverrideEnabled() & Override()
    return
  BoilingMine()
