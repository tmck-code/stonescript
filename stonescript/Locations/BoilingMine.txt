var log = import SS/Log
var strats = import SS/Strats

func Override()
  ?item.GetCooldown("wand_aether") = 0 & item.CanActivate("wand_aether")
    equipR calamity wand
    equipL calamity wand +5
    activate L
    activate R
    return true
  :?foe = boss | foe.count > 5
    activate potion
    return true
  equipR calamity wand
  equipL calamity wand +5
  return true


func BoilingMine()
  log.LogStrat(0, "BoilingMine", "#green")

  ?Override() = true
    return

  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss
    ?strats.MaskOfTheLost() = true
      return
    ?foe.distance > 12
      equip aether crossbow
      return
    ?foe.distance < 16
      activate potion
      equipL poison wand
      equipR aether wand
      activate L
      return
    equipL aether wand
    equipR aether wand *8
    return

  ?strats.Speedy()
    return

  ?strats.UseAvailableTwoHandedWeapon()
    return

  ?((foe = slave_master | foe = elemental) & foe.distance > 10)
    equip aether crossbow
    return

  ?foe.distance > 12
    equip aether crossbow
    return

  // ?strats.Dash()
  //   return

  ?foe.distance > 5
    equipL aether wand
    equipR vigor wand
    return

  equipL big aether sword
  equipR compound shield

?loc = mine
  BoilingMine()
