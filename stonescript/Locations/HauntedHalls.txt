var log = import SS/Log
// var info = import Data/Info
var strats = import SS/Strats

var BOSS = false
var i = 0

func Override()
  equipL vigor wand
  ?foe = boss | foe.distance <= 20
    equipR poison wand
  :
    ?foe.distance > 10
      equipR tri
  ?foe = boss
    ?strats.UseWandAbility("reset", "L")
      return true
    :
      ?strats.UseWandAbility("plague", "R")
        return true
  return true

func HauntedHalls()
  ?loc.loop
    BOSS = false
  log.LogStrat(4, "HauntedHalls", "#green")

  ?Override() = true
    return

  // ?strats.canUseBladeDS(2)
  //   strats.useBladeDS()
  //   return
  
  ?strats.MaskOfTheLost()
    return

  ?foe = boss
    BOSS = true
    ?foe.distance > 8 & foe ! ghost
      equip vigor crossbow
      return

    equipL vigor wand
    equipR poison wand
    return

  ?strats.Speedy()
    return

  ?foe.distance > 12
    ?foe = ghost
      ?strats.UseWandAbility("reset", "L")
        return
    :
      equip vigor crossbow
      return

  ?foe.count < 8 & foe ! ghost
    ?strats.Dash()
      return

  ?foe.distance > 10
    equipL tri
    equipR vigor wand
    return

  equipL vigor wand
  equipR poison wand

?loc = halls
  HauntedHalls()
