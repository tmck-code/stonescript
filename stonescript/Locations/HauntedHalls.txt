var log = import SS/Log
// var info = import Data/Info
var strats = import SS/Strats

var BOSS = false
var i = 0

func Override()
  equipL frost wand +10
  equipR vigor wand
  ?foe = boss | foe.count > 5
    activate potion
  ?item.GetCooldown("wand_vigor") = 0 & item.CanActivate()
    activate R
    return true
  return false

func HauntedHalls()
  ?loc.loop
    BOSS = false
  log.LogStrat(4, "HauntedHalls", "#green")

  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss
    //  | BOSS = true
    BOSS = true
    ?foe.distance > 25 & foe ! ghost
      equip vigor crossbow
      return

    equipL vigor wand
    equipR talisman broken
    return

  ?strats.Speedy()
    return

  ?Override() = true
    return

  ?foe.distance > 12 & foe ! ghost
    equip vigor crossbow
    return
  
  ?foe.distance > 5
    equipL vigor wand
    equipR poison wand
    ?item.GetCooldown("wand_vigor") = 0 & item.CanActivate() & foe.count > 5
      activate potion
      activate L
    equipL vigor wand
    return

  ?foe ! immune_to_physical & foe.distance <= 5
    strats.UseWeaponIfReady("heavy_hammer", 10)
    equipL vigor hammer
    equipR compound shield
    return

  equipL poison wand
  equipR vigor wand

?loc = halls
  HauntedHalls()
