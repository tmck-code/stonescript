var log    = import SS/Log
var strats = import SS/Strats

var needShield = false

func QuickShield()
  // about to be hit! equip the shield
  ?foe.state = 33 & foe.time > 10
    equip compound shield
    needShield = true
    return true
  // was just hit a few frames ago, equip the shield
  // for 2 frames during the next state, just in case (?)
  ?needShield = true & foe.time < 2
    equip compound shield
    return true
  // otherwise, disable the shield flag
  :
    needShield = false
    return false

func Override()
  equipL fire wand *9
  equipR fire wand *10
  ?item.GetCooldown("wand_fire") = 0 & item.CanActivate("wand_fire")
    activate R
    activate L
    return true
  return true

func IcyRidge()
  log.LogStrat(4, "IcyRidge", "#green")

  ?Override()
    return

  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss
    ?strats.Speedy()
      return

    ?QuickShield()
      return

    ?foe.distance > 12
      strats.UseWeaponIfDist("fissure", 18)
      equip fire crossbow
      return

    ?foe.distance <= 5
      strats.UseAvailableTwoHandedWeapon()
      equipL fire hammer
      equipR compound shield
      return

    equipL fire wand *9
    equipR fire wand *10
    ?item.GetCooldown("wand_fire") = 0 & item.CanActivate("wand_fire")
      activate R
      activate L
    return

  ?strats.Speedy()
    return

  ?foe = ice_elemental | foe.distance > 10
    equip fire crossbow
    return

  ?strats.Dash()
    return

  :?foe.distance <= 5
    strats.UseAvailableTwoHandedWeapon()
    equipL fire hammer
    equipR compound shield
    return

  equipL fire wand *9
  equipR fire wand *10
  ?item.GetCooldown("wand_fire") = 0 & item.CanActivate("wand_fire")
    activate R

?loc = icy_ridge
  IcyRidge()
