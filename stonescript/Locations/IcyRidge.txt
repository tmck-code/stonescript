var log    = import SS/Log
var strats = import SS/Strats

func IcyRidge()
  log.LogStrat(4, "IcyRidge", "#green")
  ?strats.canUseBladeDS(2)
    strats.useBladeDS()

  ?foe = boss
    ?foe.distance > 12
      strats.UseWeaponIfDist("fissure", 18)
      equip fire crossbow
      return

    ?foe.distance <= 5
      strats.UseAvailableTwoHandedWeapon()
      equipL fire hammer
      equipR compound shield
      return

    equipL fire wand *9
    equipR fire wand *10
    return

  ?strats.Speedy()
    return

  ?foe = ice_elemental | foe.distance > 10
    equip fire crossbow
    return

  ?strats.Dash()
    return

  :?foe.distance <= 5
    strats.UseAvailableTwoHandedWeapon()
    equipL fire hammer
    equipR compound shield
    return

  equipL fire wand *9
  equipR fire wand *10

?loc = icy_ridge
  IcyRidge()
