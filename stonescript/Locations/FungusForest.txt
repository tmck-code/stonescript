var log = import SS/Log
var strats = import SS/Strats

func FungusForest()
  log.LogStrat(4, "FungusForest", "#green")

  ?strats.canUseBladeDS(1)
    strats.useBladeDS()

  ?foe = boss
    ?strats.Speedy()
      return

    ?foe.distance >= 5
      equip poison crossbow
      return

    ?strats.Dash()
      return

    ?foe.distance <= 3 & item.GetCooldown("heavy_hammer") = 0 & item.CanActivate()
      equip heavy hammer +10
      activate R
      equipR compound shield
      equipL poison sword
      return

    equipR compound shield
    equipL poison sword
    return

  ?strats.Speedy()
    return

  ?foe.distance >= 17
    equip aether crossbow
    return

  ?strats.Dash()
    return

  ?foe.distance >= 10
    equipR poison wand
    equipL fire wand
    return

  equipL big poison sword
  equipR compound

?loc = fungus_forest
  FungusForest()
