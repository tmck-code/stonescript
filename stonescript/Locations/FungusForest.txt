var log = import SS/Log
var strats = import SS/Strats

func Override()
  equipL poison wand +11
  equipR poison wand +10
  ?foe = boss & foe = phase2
    ?strats.UseWandAbility("plague", "L")
      return true
  ?foe.distance > 20 & foe ! boss
    equipR tri
    return true
  return true

func FungusForest()
  log.LogStrat(4, "FungusForest", "#green")

  ?Override() = true
    return

  // ?strats.canUseBladeDS(3)
  //   strats.useBladeDS()

  ?foe = boss
    equip poison crossbow
    return

  ?strats.Speedy()
    return
  
  ?strats.UseWandAbility("plague", "L")
    return

  ?foe.distance > 12
    equip poison crossbow
    return

  // ?strats.Dash()
  //   return

  ?foe.distance >= 5
    equipR poison wand
    ?item.GetCooldown("wand_poison") = 0 & item.CanActivate() & foe.count > 5
      activate potion
      activate R
    equipL fire wand
    return

  equipL poison sword
  equipR compound

?loc = fungus_forest
  FungusForest()
