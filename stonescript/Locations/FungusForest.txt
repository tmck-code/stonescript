var log = import SS/Log
var strats = import SS/Strats

func Override()
  ?item.GetCooldown("wand_plague") = 0 & item.CanActivate()
    equipL poison wand +11
    equipR poison wand +10
    activate potion
    activate R
    activate L
    return true
  ?foe = boss | foe.count > 5
    activate potion
    return true
  return true


func FungusForest()
  log.LogStrat(4, "FungusForest", "#green")

  ?Override() = true
    return

  ?strats.canUseBladeDS(3)
    strats.useBladeDS()

  ?foe = boss
    ?strats.Speedy()
      return

    ?foe.distance >= 3
      ?item.GetCooldown("wand_poison") = 0 & item.CanActivate()
        activate potion
        equipR poison wand
        activate R
      equip poison crossbow
      return

    ?foe.distance <= 3 & item.GetCooldown("heavy_hammer") = 0 & item.CanActivate()
      equip heavy hammer
      activate R
      equipR compound shield
      equipL poison sword
      return

    equipR compound shield
    equipL poison sword
    return

  ?strats.Speedy()
    return

  ?item.GetCooldown("wand_poison") = 0 & item.CanActivate()
    activate potion
    equipR poison wand
    equipL fire wand
    activate R
    return

  ?foe.distance > 12
    equip poison crossbow
    return

  // ?strats.Dash()
  //   return

  ?foe.distance >= 5
    equipR poison wand
    ?item.GetCooldown("wand_poison") = 0 & item.CanActivate() & foe.count > 5
      activate potion
      activate R
    equipL fire wand
    return

  equipL poison sword
  equipR compound

?loc = fungus_forest
  FungusForest()
