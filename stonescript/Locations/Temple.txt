var log = import SS/Log
var strats = import SS/Strats

func Temple()
  log.LogStrat(4, "Temple", "#green")

  ?strats.canUseBladeDS(5)
    strats.useBladeDS()

  ?foe = boss
    strats.UseAvailableTwoHandedWeapon()
    equip ice crossbow
    return

  ?strats.Speedy()
    return

  ?foe.distance > 12
    equip ice crossbow
    return

  ?strats.Dash()
    return

  ?foe.distance > 5
    equipR ice wand
    equipL poison wand
  :
    strats.UseAvailableTwoHandedWeapon()
    equipR ice wand
    equipL ice hammer


?loc = temple
  Temple()
