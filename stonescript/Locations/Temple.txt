var log = import SS/Log
var strats = import SS/Strats

func Temple()
  log.LogStrat(4, "Temple", "#green")

  ?strats.canUseBladeDS(5)
    strats.useBladeDS()

  ?foe = boss
    strats.UseAvailableTwoHandedWeapon()
    equipL poison wand
    equipR ice wand
    return

  ?strats.Speedy()
    return

  ?foe.distance > 12
    equip ice crossbow
    return

  ?strats.Dash()
    return

  strats.UseAvailableTwoHandedWeapon()
  equipR ice wand
  equipL poison wand

?loc = temple
  Temple()
