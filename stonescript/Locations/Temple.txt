var log = import SS/Log
var strats = import SS/Strats


func Temple()
  log.LogStrat(1, "Temple", "#green")

  ?strats.canUseBladeDS(3)
    strats.useBladeDS()

  ?foe = boss
    ?foe = acronian_cultist
      ?strats.Speedy()
        return
      :
        equipL ice sword
        equipR ice hammer
        return
    activate potion
    equip ice crossbow
    return

  ?strats.Speedy()
    return

  ?foe.distance < 22 & foe.distance > 8
    equip ice crossbow
    return

  // ?strats.Dash()
  //   return

  ?foe.distance > 2
    equipL vigor wand
    equipR ice wand
  :
    equipL ice sword
    equipR ice wand

?loc = temple
  Temple()
