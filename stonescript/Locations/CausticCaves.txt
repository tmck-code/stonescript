  MAX_DISTANCE = MAX_DISTANCE + 1
  BUTTON_MORE_MAX_DISTANCE.text = string.Format("{0}>", MAX_DISTANCE)
  BUTTON_LESS_MAX_DISTANCE.text = string.Format("<{0}", MAX_DISTANCE)
