var log = import SS/Log
var strats = import SS/Strats

func Rocky()
  log.LogStrat(4, "Rocky", "#green")
  equipR vigor wand

  ?!strats.BeFast(true, false)
    ?foe.buffs.string = fire
      equipL aether wand
    :?foe.buffs.string = aether
      equipL ice wand
    :?foe.buffs.string = ice
      equipL poison wand
    // :?foe.buffs.string = poison
    :
      equipL fire wand

?loc = rocky
  Rocky()
