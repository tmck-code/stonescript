var CANCEL_ATTACK_ANIMATION = false

func _ToggleRemoteCache(keybind)
  ?key = keybind
    sys.cacheRemoteFiles = !sys.cacheRemoteFiles
    >0,10,Remote caching @sys.cacheRemoteFiles@

// Use the "up" key to toggle remote caching of scripts on & off
func ToggleRemoteCache()
  _ToggleRemoteCache(upBegin)

func CancelAttackAnimation()
  ?item.left.state = 3
    equipL *0 wand //trash
    equipL @item.left@
  ?item.right.state = 3
    equipR *0 shield //trash
    equip @item.right@

func _ToggleCancelAttackAnimation(keybind)
  ?key = keybind
    CANCEL_ATTACK_ANIMATION = !CANCEL_ATTACK_ANIMATION
    >0,10,CANCEL_ATTACK_ANIMATION: @CANCEL_ATTACK_ANIMATION@
    ?CANCEL_ATTACK_ANIMATION
      CancelAttackAnimation()

// Use the "down" key to toggle cancelling the attack animation
func ToggleCancelAttackAnimation()
  _ToggleCancelAttackAnimation(downBegin)

// hold shift to show weapons
func _ShowWeapons(keybind)
  ?key = keybind
    >`0,10,#yellow,@item.left.id@
    >`0,11,#yellow,@item.right.id@

func ShowWeapons()
  _ShowWeapons(ability1)
